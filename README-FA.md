## اینستاگرام

```javascript
const {instagram} = require("majidapi/modules/social");

// دانلود با لینک
instagram({
    method: "download",
    url: "(Post-Reels-Story-Profile-Highlight) URL"
}).then(console.log);

// دریافت پروفایل
instagram({
    method: "profile",
    username: "instagram"
}).then(console.log);

// دریافت استوری ها
instagram({
    method: "stories",
    username: "instagram"
}).then(console.log);

// دریافت هایلایت ها
instagram({
    method: "highlights",
    username: "instagram"
}).then(console.log);

// دریافت هایلایت
instagram({
    method: "highlight",
    highlightId: "highlight:id"
}).then(console.log);
```
